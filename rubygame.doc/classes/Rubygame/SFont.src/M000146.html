    def render(string)
      size = [self.string_width(string),self.height]
      render = Rubygame::Surface.new(size)
      x = 0
      string.each_byte { |glyph| 
        blit_glyph("%c"%[glyph],render,[x,0])
        x += @glyphs["%c"%[glyph]].width
      }
      return render
    end
