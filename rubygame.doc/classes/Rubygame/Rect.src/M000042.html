        def initialize(*argv)
                case argv.length
                when 1
                        if argv[0].kind_of? Array; super(argv[0])
                        elsif argv[0].respond_to? :rect; super(argv[0])
                        end
                when 2
                        super(argv[0].concat(argv[1]))
                when 4
                        super(argv)
                end
                return self
        end
