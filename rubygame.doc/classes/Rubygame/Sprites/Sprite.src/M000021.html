                        def collide_sprite?(sprite)
                                if sprite.respond_to?(:col_rect)
                                        return self.col_rect.collide_rect?(sprite.col_rect)
                                elsif sprite.respond_to?(:rect)
                                        return self.col_rect.collide_rect?(sprite.rect)
                                else
                                        return nil
                                end
                        end
