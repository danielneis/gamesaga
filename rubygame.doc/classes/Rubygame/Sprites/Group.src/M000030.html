                        def collide_group(group, killa=false, killb=false)
                                sprites = {}
                                self.each { |sprite|
                                        sprites[sprite] = sprite.collide_group(group)
                                }
                                if killa
                                        sprites.each_key { |sprite| sprite.kill }
                                end
                                if killb
                                        sprites.each_value { |sprite| sprite.kill }
                                end
                        end
