                def allow(*types)
                        added = (Rubygame::ALL_EVENT_CLASSES - types.flatten()) & @@allowed
                        for type in types.flatten()
                                if type.kind_of? Class
                                        if not @@allowed.include? type
                                                @@allowed.push(type)
                                        end
                                elsif type.kind_of? Event
                                        @@allowed.push(type.class)
                                end
                        end
                        return added
                end
