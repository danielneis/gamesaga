                def post(*events)
                        update_pending() # so posted events will appear after SDL events
                        not_added = []
                        for event in events.flatten()
                                if event.kind_of? Event and @@allowed.include?(event.class)
                                        @@pending.push(event)
                                else not_added.push(event)
                                end
                        end
                        return not_added
                end
