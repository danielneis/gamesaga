        def Rubygame.key2str( sym, mods )
                if (mods.include? K_LSHIFT) or (mods.include? K_RSHIFT)
                        return (Rubygame::Key::KEY2UPPER[sym]\
                                or Rubygame::Key::KEY2ASCII[sym] or "")
                else
                        return (Rubygame::Key::KEY2LOWER[sym]\
                                or Rubygame::Key::KEY2ASCII[sym] or "")
                end
        end
